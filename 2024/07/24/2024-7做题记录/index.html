    B - MST [场切]

    好像有一个和场上做法都不一样的做法，简单讲一下。

    考虑所有给出点集构成的导出子图的边数和。若给出点集大小 ，那么导出子图边数为 ，否则边数至多为 。那么大导出子图的总边数至多为 , 小导出子图的总边数也是至多的（由于二次函数的凸性）。

    这样一来，如果我们能够快速的得到所有导出子图有哪些边，我们就可以解决这道题。

    我们直接进行一个力的暴，用一个bitset存每个点被包含在哪些询问中。利用Kruskal，我们将所有边先排序，然后尝试把每条边加入包含他的导出子图中。那么一条边被包含的导出子图，恰是两点bitset的并集。

    复杂度为

    代码 
    int n,m,q;
int k[maxn];
vector<int> s[maxn];
unordered_map<ll ,int> fa;
ll key(int a,int b){return ((ll)a)<<32|(ll)b;}
int find(int i,int x){
    if(fa[key(i,x)]==x)return x;
    else return fa[key(i,x)]=find(i,fa[key(i,x)]);
}
struct edge{
    int u,v;
    ll w;
}e[maxn];
bitset<100001> b[maxn];
ll ans[maxn];
int main(){
    read(n),read(m),read(q);
    for(int i=1;i<=m;i++)
        read(e[i].u),read(e[i].v),read(e[i].w);
    sort(e+1,e+m+1,[&](const edge &a, const edge &b){
        return a.w<b.w;
    });
    for(int i=1;i<=q;i++){
        read(k[i]);
        s[i].resize(k[i]+1);
        for(int j=1,x;j<=k[i];j++){
            read(x);fa[key(i,x)]=x;
            b[x].set(i);
            s[i][j]=x;
        }
    }
    for(int i=1;i<=m;i++){
        int u=e[i].u,v=e[i].v;
        bitset<100001> cur = b[u]&b[v];
        for(int pos=cur._Find_first();pos!=100001;pos=cur._Find_next(pos)){
            if(find(pos,u)!=find(pos,v)){
                fa[key(pos,find(pos,u))]=find(pos,v);
                ans[pos]+=1ll*e[i].w;
            }
        }
    }
    for(int i=1;i<=q;i++){
        int cnt=0;
        for(int j=1;j<=k[i];j++)
            if(find(i,s[i][j])==s[i][j])cnt++;
        if(cnt>1)printf("-1\n");
        else printf("%lld\n",ans[i]);
    }
    return 0;
}
